## Usage

### Basic Usage

```tsx
import { ChatInput, ChatInputField, ChatInputActions } from "@/components/prismui/chat-input"
import { Button } from "@/components/ui/button"
import { SendIcon } from "lucide-react"

export default function Example() {
  const [value, setValue] = useState("")
  const [isLoading, setIsLoading] = useState(false)

  const handleSubmit = () => {
    setIsLoading(true)
    // Your submit logic here
    setIsLoading(false)
  }

  return (
    <ChatInput>
      <ChatInputField
        value={value}
        onChange={setValue}
        onSubmit={handleSubmit}
        isLoading={isLoading}
      />
      <ChatInputActions>
        <Button
          size="icon"
          className="rounded-full"
          disabled={!value || isLoading}
          onClick={handleSubmit}
        >
          {isLoading ? "..." : <SendIcon className="h-4 w-4" />}
        </Button>
      </ChatInputActions>
    </ChatInput>
  )
}
```

## Props

### ChatInputProps
| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| className | string | - | Additional CSS classes for the root container |
| children | ReactNode | - | Child components (ChatInputField and ChatInputActions) |

### ChatInputFieldProps
| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| value | string | - | The current value of the input field |
| onChange | (value: string) => void | - | Callback when input value changes |
| onSubmit | () => void | - | Optional callback when submit is triggered |
| isLoading | boolean | false | Whether the input is in a loading state |
| className | string | - | Additional CSS classes for the input field |

### ChatInputActionsProps
| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| className | string | - | Additional CSS classes for the actions container |
| children | ReactNode | - | Action buttons or other components |

### ChatInputLegacyProps
| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| value | string | - | The current value of the input field |
| onChange | (value: string) => void | - | Callback when input value changes |
| onSubmit | () => void | - | Optional callback when submit is triggered |
| isLoading | boolean | false | Whether the input is in a loading state |
| className | string | - | Additional CSS classes for the root container |
| inputClassName | string | - | Additional CSS classes for the input field |
